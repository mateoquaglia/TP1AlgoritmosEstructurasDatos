#include "Pila.h"
#include<string>
#include<string.h>
#include<iostream>
#include <stdio.h>
#include <stdlib.h>
using namespace std;
using std::string;



Pila::Pila(Lista* lVariables)
{
    listaVariables=lVariables;
    cabeza="";
    next=NULL;
    esBool_o_Entera=false;
}
Pila::Pila()
{
    cabeza="";
    next=NULL;
    esBool_o_Entera=false;
    listaVariables=NULL;

};
void Pila::apilar(string p)
{

    Pila *nuevoNext=new Pila();

    nuevoNext->setCabeza(this->getCabeza());

    nuevoNext->setNext(this->getNext());

    this->setCabeza(p);

    this->setNext(nuevoNext);

};
string Pila::tope()
{
    return this->getCabeza();
};
void Pila::desapilar(){
    if(!pilavacia()){
        Pila *next=this->getNext();
        if(next!=NULL){
            this->setCabeza(next->getCabeza());
            this->setNext(next->getNext());
        }else{
            this->setCabeza("");
        }
    }
};
bool Pila::pilavacia()
{
    if(cabeza=="" && next==NULL)
    {
        return true;
    }
    return false;
}
void Pila::setCabeza(string s)
{
    cabeza=s;
}
string Pila::getCabeza()
{
    return cabeza;
}
void Pila::setNext(Pila* n)
{
    next=n;
}
Pila* Pila::getNext()
{
    return next;
}
