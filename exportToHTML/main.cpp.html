#include <iostream>
#include <vector>
#include <stack>
#include <queue>
#include "Fruta.h"
#include "Cajon.h"
#include "Pedidos.h"
#include "Stock.h"

using namespace std;

int main()
{

    // Crear instancias de las frutas
    Fruta manzana("Manzana", 10);
    Fruta pera("Pera", 12);
    Fruta banana("Banana", 15);

    // Crear instancias de los cajones
    Cajon cajon1("A001", "Manzana", 10);
    Cajon cajon2("A002", "Pera", 8);
    Cajon cajon3("A003", "Banana", 16);

    // Crear un vector de cajones
    vector<Cajon> cajones;
    cajones.push_back(cajon1);
    cajones.push_back(cajon2);
    cajones.push_back(cajon3);

    // Crear instancias de los pedidos
    Pedido pedido1("Juan", "Manzana", 20, false, true);
    Pedido pedido2("Pedro", "Pera", 10, true, false);

    // Crear una cola de pedidos
    queue<Pedido> pedidos;
    pedidos.push(pedido1);
    pedidos.push(pedido2);

    // Crear una instancia del stock
    Stock stock(cajones);

    // Actualizar el stock con los cajones creados anteriormente
    stock.actualizarStock();

    // Imprimir el stock actual
    stock.imprimirStock();

    // Procesar los pedidos
    stock.procesarPedidos(pedidos);

    // Imprimir los pedidos procesados
    stock.imprimirPedidos();

    return 0;
}
