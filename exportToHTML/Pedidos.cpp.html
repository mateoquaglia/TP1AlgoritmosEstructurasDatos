string Pedido::determinarTipoPedido(Pedido pedido)
{
    if(pedido.getPesoPedido() < 20){
        esMinorista() == true;
    }
    if(pedido.getPesoPedido() == 20 || (pedido.getPesoPedido() >20 && pedido.getPesoPedido()%20 == 0)){
        esMayorista() == true;
    }

}
void Pedido::addPedido(Pedido pedido)
{
    if(pedido.esMayorista()){
        pedido.addPedido(ColaMayorista);
    }else{
        pedido.addPedido(ColaMinorista);
    }

}
