#ifndef CAJON_H
#define CAJON_H
#include<string>
#include<iostream>
using namespace std;
class Cajon
{
private:
    int id;
    string tipoFruta;
    int cantidadFruta;
public:
    Cajon(int id, Fruta* fruta, int cantidad)
    {
        this->id = id;
        tipoFruta = fruta.getNombre();
        cantidadFruta = cantidad;

    }

    int getID();
    string getTipoFruta();
    int getCantidadFruta();
    void setCantidadFruta(int cantidad);
};


#endif
