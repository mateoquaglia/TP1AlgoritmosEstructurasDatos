#include "Cola.h"

Cola::Cola() {
    lista = new Lista();
}

Cola::~Cola() {
    delete lista;
}

void Cola::encolar(string valor) {
    lista->agregarAlFinal(valor);
}

void Cola::desencolar() {
    lista->eliminarAlInicio();
}

string Cola::frente() {
    return lista->getInicio()->dato;
}

bool Cola::colaVacia() {
    return lista->estaVacia();
}
